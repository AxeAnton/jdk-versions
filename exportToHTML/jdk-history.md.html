78   ## Основные изменения в Java 14
79   
80   1. Switch выражения
81   
82   
83          String season = "";
84          int month = 2;
85   
86           season = switch (month) {
87               case 12, 1, 2 -> "Зима";
88               case 3, 4, 5 -> "Весна";
89               case 6, 7, 8 -> "Лето";
90               case 9, 10, 11 -> "Осень";
91               default -> "Ошибка ввода";
92           };
93   
94   2. Switch выражения и ключевое слово yield
95   
96   
97           String season = "";
98           int month = 2;
99   
100          season = switch (month) {
101              case 12, 1, 2 -> {
102                  System.out.println(month);
103                  yield "Зима";
104              }
105              case 3, 4, 5 -> {
106                  System.out.println(month);
107                  yield "Весна";
108              }
109              case 6, 7, 8 -> {
110                  System.out.println(month);
111                  yield "Лето";
112              }
113              case 9, 10, 11 -> {
114                 System.out.println(month);
115                 yield "Осень";
116              }
117              default -> "Ошибка ввода";
118          };
